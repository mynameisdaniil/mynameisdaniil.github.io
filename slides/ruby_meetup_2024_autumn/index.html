  def scrape(base_url, range, dir) do
    parent = self()
    pids = for i <- range do
      spawn(fn ->
        case HTTPoison.get!("#{base_url}/ru/feed/page#{i}/") do
          %{status_code: 200, body: body} ->
            File.write!("#{dir}/#{i}.html", body)
            send(parent, {:ok, self()})
          _otherwise ->
            {:ok, self()}
        end
      end)
    end

    wait_for(pids)
  end
```

  def parse(dir, output) do
    json = File.ls!(dir)
           |> Enum.map(fn filename ->
             File.read!(Path.join(dir, filename))
           end)
           |> Enum.map(&Floki.parse_document!/1)
           |> Flow.from_enumerable()
           |> Flow.flat_map(fn document ->
             ...
             end)
             |> Enum.reject(&(&1 == :skip))
             |> Enum.to_list()
           end)
           |> Enum.to_list()
           |> :json.encode()
    File.write(output, json, [:binary, :raw])
  end
```

  def stat_by_weekday(filename, field) do
    file = File.read!(filename)
    data = :json.decode(file)

    stats = data
    |> Enum.reduce(%{}, fn item, acc ->
      date = NaiveDateTime.from_iso8601!(item["time"])
             |> DateTime.from_naive!("Europe/Moscow", Tz.TimeZoneDatabase)
      weekday = Date.day_of_week(date, :sunday)
      days = [:sun, :mon, :tue, :wed, :thu, :fri, :sat]
      day = days |> Enum.at(weekday - 1)
      acc |> Map.update(day, item[field], fn current ->
        current + item[field]
      end)
    end)

    {_, max_value} = stats |> Enum.max_by(fn {_k, v} -> v end)

    stats |> Enum.map(fn {k, v} -> {k, v/max_value} end)
  end
